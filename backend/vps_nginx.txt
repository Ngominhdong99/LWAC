server {
    server_name 103.121.91.177 minhdong-edu.io.vn www.minhdong-edu.io.vn;

    root /opt/LWAC/frontend/dist;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location /api/ {
        proxy_pass http://127.0.0.1:8000/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        client_max_body_size 100M;
    }

    location /static/ {
        alias /opt/LWAC/backend/static/;
        expires 30d;
        add_header Cache-Control public, ;
    }

    location /uploads/ {
        alias /opt/LWAC/backend/uploads/;
    }

    listen 443 ssl;
    ssl_certificate /etc/letsencrypt/live/minhdong-edu.io.vn/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/minhdong-edu.io.vn/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}
server {
    if ($host = www.minhdong-edu.io.vn) {
        return 301 https://$host$request_uri;
    }
    if ($host = minhdong-edu.io.vn) {
        return 301 https://$host$request_uri;
    }
    listen 80;
    server_name 103.121.91.177 minhdong-edu.io.vn www.minhdong-edu.io.vn;
    return 404;
}
